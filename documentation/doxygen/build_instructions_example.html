Build Instructions for example

  1. Add the dependencies to configure/RELEASE. 

    PROCSERVCONTROL=/dls_sw/prod/R3.14.11/support/procServControl/1-2
ASYN=/dls_sw/prod/R3.14.11/support/asyn/4-17
BUSY=/dls_sw/prod/R3.14.11/support/busy/1-3dls5
SEQ=/dls_sw/prod/R3.14.11/support/seq/2-1-2dls2

  2. Add the DBD dependencies to src/Makefile 

    example_DBD += base.dbd
example_DBD += asyn.dbd
example_DBD += drvAsynIPPort.dbd
example_DBD += busySupport.dbd
example_DBD += procServControl.dbd

  3. Add the LIBS dependencies to src/Makefile 

    example_LIBS += procServControl
example_LIBS += busy
example_LIBS += seq
example_LIBS += pv
example_LIBS += asyn

  4. Use the template files to add records to the database. 

    
# Macros:
#  P            Device prefix
#  PORT         Asyn port name
#  SHOWOUT      Set this to 0 to disable the IOC output display
#  MANUALSTART  Set this to # if the IOC is manually started. This changes the severity of STATUS
#  name         Object and gui association name
file $(PROCSERVCONTROL)/db/procServControl.template
{
pattern { P, PORT, SHOWOUT, MANUALSTART, name }
    { "BLxxI-TS-IOC-01", "port1", "1", "", "BLxxI-TS-IOC-01" }
    { "BLxxI-TS-IOC-02", "port2", "1", "", "BLxxI-TS-IOC-02" }
}

  5. Add the startup commands to st.cmd 
    
# Loading libraries
# -----------------

# Device initialisation
# ---------------------

cd "$(TOP)"

dbLoadDatabase "dbd/example.dbd"
example_registerRecordDeviceDriver(pdbbase)

drvAsynIPPortConfigure("port1", "localhost:7001", 100, 0, 0)

drvAsynIPPortConfigure("port2", "localhost:7002", 100, 0, 0)

seq(procServControl,"P=BLxxI-TS-IOC-01")

seq(procServControl,"P=BLxxI-TS-IOC-02")
